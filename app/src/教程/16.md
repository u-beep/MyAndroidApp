# 一、什么是 File 存储？
大白话：
**就是在手机里创建一个 txt 文本，把文字写进去，永久保存。**

特点：
- 保存文字、笔记、日志
- 比 SP 能存**更多内容**
- 卸载 APP 才会删除
- 简单、安全、私密

# 🚀 第 3 步：核心代码 —— 文件保存与读取
打开 `MemoActivity.kt`
**直接复制全覆盖**

```kotlin
import android.os.Bundle
import android.widget.Button
import android.widget.EditText
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import java.io.FileInputStream
import java.io.FileOutputStream

class MemoActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_memo)

        val etMemo = findViewById<EditText>(R.id.et_memo)
        val btnSave = findViewById<Button>(R.id.btn_save)

        // 一、打开页面时 → 自动读取文件
        readMemo(etMemo)

        // 二、点击保存 → 写入文件
        btnSave.setOnClickListener {
            val content = etMemo.text.toString().trim()
            if (content.isEmpty()) {
                Toast.makeText(this, "内容不能为空", Toast.LENGTH_SHORT).show()
                return@setOnClickListener
            }

            saveMemo(content)

            Toast.makeText(this, "保存成功！文件路径：\n${filesDir}/memo.txt", Toast.LENGTH_LONG).show()
        }
    }

    // ====================== 核心方法 ======================

    // 1. 保存到文件
    private fun saveMemo(content: String) {
        try {
            // 打开文件输出流
            val fos: FileOutputStream = openFileOutput("memo.txt", MODE_PRIVATE)
            fos.write(content.toByteArray())
            fos.flush()
            fos.close()
        } catch (e: Exception) {
            e.printStackTrace()
        }
    }

    // 2. 读取文件
    private fun readMemo(etMemo: EditText) {
        try {
            // 打开文件输入流
            val fis: FileInputStream = openFileInput("memo.txt")
            val byteArray = ByteArray(fis.available())
            fis.read(byteArray)
            fis.close()

            // 转为字符串显示到输入框
            val content = String(byteArray)
            etMemo.setText(content)
        } catch (e: Exception) {
            e.printStackTrace()
        }
    }
}
```

# 四、最重要的 3 个方法（背下来）
### 1. 保存文件
```kotlin
openFileOutput("文件名", 模式)
write(内容)
```

### 2. 读取文件
```kotlin
openFileInput("文件名")
read()
```

### 3. 模式
- `MODE_PRIVATE`：**只有自己 APP 能访问，最安全**
