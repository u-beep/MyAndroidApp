# 🚀 第 1 步：创建数据库帮助类（建库、建表）
新建 `MySQLiteHelper.kt`
```kotlin
import android.content.Context
import android.database.sqlite.SQLiteDatabase
import android.database.sqlite.SQLiteOpenHelper

class MySQLiteHelper(context: Context) : SQLiteOpenHelper(
    context,
    "user_db.db", // 数据库名字
    null,
    1 // 版本号
) {

    // 第一次安装APP时调用：创建表
    override fun onCreate(db: SQLiteDatabase) {
        // 创建用户表
        val sql = """
            CREATE TABLE user_table (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                account TEXT UNIQUE,
                pwd TEXT,
                sex TEXT,
                hobby TEXT
            )
        """.trimIndent()
        db.execSQL(sql)
    }

    override fun onUpgrade(db: SQLiteDatabase?, oldVersion: Int, newVersion: Int) {}
}
```

# 🚀 第 2 步：创建用户数据操作类（UserDao.kt）
**增、删、改、查 全部在这里！**

```kotlin
import android.content.ContentValues
import android.content.Context
import android.database.sqlite.SQLiteDatabase

class UserDao(context: Context) {
    private val db: SQLiteDatabase

    init {
        val helper = MySQLiteHelper(context)
        db = helper.writableDatabase
    }

    // 1. 增加：注册用户
    fun addUser(account: String, pwd: String, sex: String, hobby: String): Boolean {
        val cv = ContentValues()
        cv.put("account", account)
        cv.put("pwd", pwd)
        cv.put("sex", sex)
        cv.put("hobby", hobby)

        val id = db.insert("user_table", null, cv)
        return id != -1L
    }

    // 2. 查询：登录用（根据账号查密码）
    fun login(account: String): String? {
        val cursor = db.query(
            "user_table",
            null,
            "account=?",
            arrayOf(account),
            null, null, null
        )

        if (cursor.moveToNext()) {
            val pwd = cursor.getString(cursor.getColumnIndexOrThrow("pwd"))
            cursor.close()
            return pwd
        }
        cursor.close()
        return null
    }

    // 3. 查询所有用户（给列表用）
    fun getAllUser(): MutableList<User> {
        val list = mutableListOf<User>()
        val cursor = db.query("user_table", null, null, null, null, null, null)

        while (cursor.moveToNext()) {
            val account = cursor.getString(1)
            val pwd = cursor.getString(2)
            val sex = cursor.getString(3)
            val hobby = cursor.getString(4)
            list.add(User(account, pwd, sex, hobby))
        }
        cursor.close()
        return list
    }

    // 4. 修改密码
    fun updatePwd(account: String, newPwd: String): Int {
        val cv = ContentValues()
        cv.put("pwd", newPwd)
        return db.update("user_table", cv, "account=?", arrayOf(account))
    }

    // 5. 删除用户
    fun deleteUser(account: String): Int {
        return db.delete("user_table", "account=?", arrayOf(account))
    }
}
```
