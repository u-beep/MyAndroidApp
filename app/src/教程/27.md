## 四、步骤3：多布局适配器 MultiMemoAdapter.kt（核心）
### 常量定义：3种Type
```kotlin
companion object{
    const val TYPE_NORMAL = 0   //普通备忘录
    const val TYPE_AD = 1       //广告
    const val TYPE_FOOT = 2     //底部加载
}
```
完整Adapter代码
```kotlin
import android.view.LayoutInflater
import android.view.ViewGroup
import androidx.recyclerview.widget.RecyclerView
import com.example.myapplication.databinding.ItemMemoNormalBinding
import com.example.myapplication.databinding.ItemMemoAdBinding
import com.example.myapplication.databinding.ItemFooterLoadBinding

class MultiMemoAdapter(var dataList:MutableList<Memo>):RecyclerView.Adapter<RecyclerView.ViewHolder>() {
    //条目点击、长按删除回调
    var itemClick:(Memo)->Unit = {}
    var longClick:(Memo)->Unit = {}
    //上拉加载：是否还有下一页
    var hasMore:Boolean = true

    //=========三种ViewHolder=========
    inner class NormalVH(val bind:ItemMemoNormalBinding):RecyclerView.ViewHolder(bind.root)
    inner class AdVH(val bind:ItemMemoAdBinding):RecyclerView.ViewHolder(bind.root)
    inner class FootVH(val bind:ItemFooterLoadBinding):RecyclerView.ViewHolder(bind.root)

    //1.根据position返回条目类型（多布局关键）
    override fun getItemViewType(position: Int): Int {
        //最后一条固定是底部脚布局
        return if(position == itemCount-1) TYPE_FOOT else dataList[position].itemType
    }

    //2.根据type创建不同布局ViewHolder
    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): RecyclerView.ViewHolder {
        val inflate = LayoutInflater.from(parent.context)
        return when(viewType){
            TYPE_NORMAL -> NormalVH(ItemMemoNormalBinding.inflate(inflate,parent,false))
            TYPE_AD -> AdVH(ItemMemoAdBinding.inflate(inflate,parent,false))
            else -> FootVH(ItemFooterLoadBinding.inflate(inflate,parent,false))
        }
    }

    //3.绑定数据，区分类型赋值
    override fun onBindViewHolder(holder: RecyclerView.ViewHolder, position: Int) {
        when(holder){
            is NormalVH ->{
                val memo = dataList[position]
                holder.bind.tvTitle.text = memo.title
                holder.bind.tvContent.text = memo.content
                holder.bind.tvTime.text = memo.createTime
                holder.itemView.setOnClickListener { itemClick(memo) }
                holder.itemView.setOnLongClickListener { longClick(memo);true }
            }
            is FootVH ->{
                //修改底部文案：无更多/加载中
                holder.bind.tvFooter.text = if(hasMore) "正在加载更多..." else "没有更多备忘录了"
            }
        }
    }

    //总数=数据数量+1条底部脚布局
    override fun getItemCount() = dataList.size + 1

    //下拉刷新：替换全量数据
    fun refreshAll(newList:MutableList<Memo>){
        dataList = newList
        notifyDataSetChanged()
    }
    //上拉加载：追加数据
    fun addMoreData(addList:MutableList<Memo>){
        val oldSize = dataList.size
        dataList.addAll(addList)
        notifyItemRangeInserted(oldSize,addList.size)
    }
}
```

## 六、步骤5：MemoListActivity完整逻辑（下拉刷新+上拉分页）
分页规则：
- PAGE_SIZE=5，每页5条数据
- page=1 下拉刷新重置page=1，清空列表
- 上拉page++，查询下一页，数据不足5条则`hasMore=false`
```kotlin
import android.content.Intent
import android.os.Bundle
import android.os.Handler
import android.os.Looper
import android.widget.Toast
import androidx.appcompat.app.AlertDialog
import androidx.appcompat.app.AppCompatActivity
import androidx.recyclerview.widget.LinearLayoutManager
import androidx.recyclerview.widget.RecyclerView
import com.example.myapplication.databinding.ActivityMemoListBinding

class MemoListActivity : AppCompatActivity() {
    private lateinit var binding:ActivityMemoListBinding
    private val dao = MemoDao(this)
    private lateinit var adapter:MultiMemoAdapter
    //分页参数
    private var page = 1
    private val PAGE_SIZE = 5
    private var isLoading = false //防止重复上拉加载
    private var layoutManager:LinearLayoutManager?=null

    companion object{
        const val KEY_MEMO_ID = "memo_id"
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivityMemoListBinding.inflate(layoutInflater)
        setContentView(binding.root)

        adapter = MultiMemoAdapter(mutableListOf())
        layoutManager = LinearLayoutManager(this)
        binding.rvMemo.adapter = adapter
        binding.rvMemo.layoutManager = layoutManager

        //=====1.下拉刷新监听=====
        binding.swipeRefresh.setOnRefreshListener {
            page = 1
            adapter.hasMore = true
            loadData(true) //true=下拉刷新
        }

        //=====2.上拉加载：RecyclerView滚动监听=====
        binding.rvMemo.addOnScrollListener(object :RecyclerView.OnScrollListener(){
            override fun onScrollStateChanged(recyclerView: RecyclerView, newState: Int) {
                super.onScrollStateChanged(recyclerView, newState)
                //静止+滑到最后一条+不在加载中+还有更多
                val lastPos = layoutManager!!.findLastVisibleItemPosition()
                val totalCount = adapter.itemCount
                if(newState == RecyclerView.SCROLL_STATE_IDLE
                    && lastPos == totalCount-1
                    && !isLoading
                    && adapter.hasMore
                    && !binding.swipeRefresh.isRefreshing){
                    page++
                    loadData(false) //false=上拉加载
                }
            }
        })

        //新增备忘录按钮
        binding.fabAdd.setOnClickListener {
            startActivity(Intent(this,EditMemoActivity::class.java).apply { putExtra(KEY_MEMO_ID,-1L) })
        }

        //条目点击编辑
        adapter.itemClick = { memo->
            startActivity(Intent(this,EditMemoActivity::class.java).apply { putExtra(KEY_MEMO_ID,memo.id) })
        }

        //长按删除弹窗
        adapter.longClick = { memo->
            AlertDialog.Builder(this)
                .setTitle("删除")
                .setMessage("确定删除${memo.title}？")
                .setPositiveButton("确定"){_,_ ->
                    dao.deleteMemo(memo.id)
                    page=1
                    loadData(true)
                }.setNegativeButton("取消",null).show()
        }

        //首次加载
        loadData(true)
    }

    //加载数据 isRefresh:true下拉刷新 false上拉追加
    private fun loadData(isRefresh:Boolean){
        isLoading = true
        //模拟网络延迟1秒
        Handler(Looper.getMainLooper()).postDelayed({
            //分页查询全部数据，手动分页截取
            val allData = dao.getAllMemo()
            val startIndex = (page-1)*PAGE_SIZE
            val endIndex = startIndex+PAGE_SIZE
            var pageData = mutableListOf<Memo>()
            if(startIndex < allData.size){
                val sub = if(endIndex>allData.size) allData.subList(startIndex,allData.size) else allData.subList(startIndex,endIndex)
                pageData = sub.toMutableList()
            }

            //随机在第二条插入广告itemType=1
            if(pageData.isNotEmpty() && pageData.size>=2){
                pageData[1] = pageData[1].copy(itemType = 1)
            }

            //判断是否还有下一页
            adapter.hasMore = endIndex < allData.size

            if(isRefresh){
                adapter.refreshAll(pageData)
            }else{
                adapter.addMoreData(pageData)
            }
            binding.swipeRefresh.isRefreshing = false //关闭刷新转圈
            isLoading = false
        },1000)
    }

    //从编辑页返回自动刷新
    override fun onResume() {
        super.onResume()
        page=1
        adapter.hasMore=true
        loadData(true)
    }
}
```

## 七、运行效果说明
1. **多布局展示**：列表第2条自动变成广告样式，底部永久显示加载脚（加载中/无更多）
2. **下拉刷新**：手指下拉列表顶部转圈，page重置1，数据重新从第一页加载，广告重新生成
3. **上拉加载**：滑到列表最底端自动请求下一页5条，不足5条底部变为「没有更多备忘录了」，停止加载
4. **新增/删除备忘录**：返回列表自动全量刷新

## 🎯今日必背4个重点
1. **多布局核心**：`getItemViewType(position)`返回类型，onCreateViewHolder根据type创建不同布局VH
2. **SwipeRefresh**：包裹RV，`setOnRefreshListener`下拉回调，`isRefreshing=false`关闭加载动画
3. **上拉加载原理**：滚动监听`findLastVisibleItemPosition()==总条目-1`，空闲状态触发加载，`isLoading`防重复请求
4. **分页逻辑**：page=1刷新重置，page++上拉，剩余数据<PAGE_SIZE → hasMore=false
