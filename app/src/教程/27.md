## 一、SP基础知识点（必背）
1. 存储路径：`/data/data/应用包名/shared_prefs/xxx.xml`，卸载APP数据清空
2. 唯一可用模式：`MODE_PRIVATE`（私有），其余模式全部废弃
3. 三步存储：获取SP→edit()获取编辑器→putXXX存值→apply()异步保存（推荐，不阻塞主线程）
4. 常用数据：String/Boolean/Int/Float/Long，**不能直接存对象、集合**

## 二、步骤1：新建SP工具类 SPUtil.kt（工具包utils）
全局统一管理SP，项目只创建一个SP文件`user_config`
```kotlin
import android.content.Context
import android.content.SharedPreferences

object SPUtil {
    //SP文件名
    private const val SP_NAME = "user_config"
    private lateinit var sp: SharedPreferences

    //初始化（Application中调用一次即可，下面附带Application代码）
    fun init(context: Context) {
        sp = context.getSharedPreferences(SP_NAME, Context.MODE_PRIVATE)
    }

    //保存字符串：账号、密码
    fun putString(key: String, value: String) {
        sp.edit().putString(key, value).apply()
    }
    fun getString(key: String, def: String = ""): String {
        return sp.getString(key, def) ?: def
    }

    //保存布尔：记住密码、自动登录、夜间模式
    fun putBoolean(key: String, value: Boolean) {
        sp.edit().putBoolean(key, value).apply()
    }
    fun getBoolean(key: String, def: Boolean = false): Boolean {
        return sp.getBoolean(key, def)
    }

    //保存int：字体大小
    fun putInt(key: String, value: Int) {
        sp.edit().putInt(key, value).apply()
    }
    fun getInt(key: String, def: Int = 16): Int {
        return sp.getInt(key, def)
    }

    //删除单个key
    fun remove(key: String) {
        sp.edit().remove(key).apply()
    }
    //清空全部SP数据
    fun clearAll() {
        sp.edit().clear().apply()
    }
}
```

## 五、步骤4：LoginActivity.kt核心逻辑（记住密码+自动登录）
### SP常量key（统一命名，防止写错）
```kotlin
companion object {
    const val KEY_ACCOUNT = "save_account"
    const val KEY_PWD = "save_pwd"
    const val KEY_REMEMBER = "is_remember"
    const val KEY_AUTO = "is_auto_login"
}
```
完整Activity代码
```kotlin
import android.content.Intent
import android.os.Bundle
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import com.example.myapplication.databinding.ActivityLoginBinding

class LoginActivity : AppCompatActivity() {
    private lateinit var binding: ActivityLoginBinding
    private val userDao = com.example.myapplication.UserDao(this)

    companion object {
        const val KEY_ACCOUNT = "save_account"
        const val KEY_PWD = "save_pwd"
        const val KEY_REMEMBER = "is_remember"
        const val KEY_AUTO = "is_auto_login"
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivityLoginBinding.inflate(layoutInflater)
        setContentView(binding.root)

        //=========页面初始化：读取SP，回填账号密码、勾选状态=========
        val isRemember = SPUtil.getBoolean(KEY_REMEMBER)
        val isAuto = SPUtil.getBoolean(KEY_AUTO)
        binding.cbRemember.isChecked = isRemember
        binding.cbAuto.isChecked = isAuto

        //记住密码开启→填充账号密码
        if (isRemember) {
            binding.etAccount.setText(SPUtil.getString(KEY_ACCOUNT))
            binding.etPwd.setText(SPUtil.getString(KEY_PWD))
        }

        //自动登录开启且记住密码→直接跳转主页
        if (isAuto && isRemember) {
            startActivity(Intent(this, com.example.myapplication.MainActivity::class.java))
            finish()
        }

        //=========登录按钮点击事件=========
        binding.btnLogin.setOnClickListener {
            val account = binding.etAccount.text.toString().trim()
            val pwd = binding.etPwd.text.toString().trim()
            //SQLite校验账号密码（沿用之前注册用户数据库）
            val loginUser = userDao.findUserByAccount(account)
            if (loginUser == null || loginUser.password != pwd) {
                Toast.makeText(this, "账号或密码错误", Toast.LENGTH_SHORT).show()
                return@setOnClickListener
            }

            //登录成功：保存勾选状态+账号密码
            val remember = binding.cbRemember.isChecked
            val auto = binding.cbAuto.isChecked
            SPUtil.putBoolean(KEY_REMEMBER, remember)
            SPUtil.putBoolean(KEY_AUTO, auto)
            //勾选记住密码才存储账号密码，否则清空
            if (remember) {
                SPUtil.putString(KEY_ACCOUNT, account)
                SPUtil.putString(KEY_PWD, pwd)
            } else {
                SPUtil.remove(KEY_ACCOUNT)
                SPUtil.remove(KEY_PWD)
            }

            //跳转首页
            startActivity(Intent(this, com.example.myapplication.MainActivity::class.java))
            finish()
            Toast.makeText(this, "登录成功", Toast.LENGTH_SHORT).show()
        }

        //注册跳转
        binding.btnReg.setOnClickListener {
            startActivity(Intent(this, com.example.myapplication.RegisterActivity::class.java))
        }
    }
}
```

```kotlin
import android.content.Intent
import android.os.Bundle
import android.widget.SeekBar
import androidx.appcompat.app.AppCompatActivity
import com.example.myapplication.databinding.ActivitySettingBinding

class SettingActivity : AppCompatActivity() {
    private lateinit var binding: ActivitySettingBinding
    private val KEY_FONT = "font_size"
    private val KEY_NIGHT = "night_mode"

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivitySettingBinding.inflate(layoutInflater)
        setContentView(binding.root)

        //读取SP配置
        val fontSize = SPUtil.getInt(KEY_FONT, 16)
        val nightOpen = SPUtil.getBoolean(KEY_NIGHT)
        binding.cbNight.isChecked = nightOpen
        binding.sbFont.progress = fontSize - 12
        binding.tvFont.text = "当前字体大小：${fontSize}sp"

        //拖动条修改字体大小保存SP
        binding.sbFont.setOnSeekBarChangeListener(object : SeekBar.OnSeekBarChangeListener {
            override fun onProgressChanged(seekBar: SeekBar?, progress: Int, fromUser: Boolean) {
                val size = progress + 12 //12~22sp
                binding.tvFont.text = "当前字体大小：${size}sp"
                SPUtil.putInt(KEY_FONT, size)
            }
            override fun onStartTrackingTouch(p0: SeekBar?) {}
            override fun onStopTrackingTouch(p0: SeekBar?) {}
        })

        //夜间模式勾选保存
        binding.cbNight.setOnCheckedChangeListener { _, isChecked ->
            SPUtil.putBoolean(KEY_NIGHT, isChecked)
        }

        //退出登录：清空自动登录，返回登录页
        binding.btnLogout.setOnClickListener {
            SPUtil.putBoolean(LoginActivity.KEY_AUTO, false)
            startActivity(Intent(this, LoginActivity::class.java))
            finish()
        }
    }
}
```

## 🎯今日必背5句话
1. SharedPreferences适合**少量配置数据持久化**，大数据用SQLite
2. `apply()`异步保存，推荐；`commit()`同步阻塞主线程，极少使用
3. 记住密码：勾选→SP存账号密码；取消勾选→remove删除key
4. 自动登录依赖「记住密码+自动登录双勾选」，缺一不可
5. SP封装工具类，项目统一管理key，便于后期维护修改
