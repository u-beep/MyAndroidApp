## 一、基础概念
- **Activity：整页容器**，一个页面窗口
- **Fragment：内嵌在Activity里的子页面**，同一屏幕多个Fragment互相切换，节省页面跳转

### 2. UserListFragment（原来的用户列表页面）
```kotlin
import android.os.Bundle
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import androidx.fragment.app.Fragment
import androidx.recyclerview.widget.LinearLayoutManager
import com.example.myapplication.User
import com.example.myapplication.UserAdapter
import com.example.myapplication.UserDao

class UserListFragment : Fragment() {
    private var userList = mutableListOf<User>()
    private lateinit var userAdapter: UserAdapter
    override fun onCreateView(inflater: LayoutInflater, container: ViewGroup?, savedInstanceState: Bundle?): View? {
        val view = inflater.inflate(R.layout.fragment_userlist,container,false)
        val rv = view.findViewById<androidx.recyclerview.widget.RecyclerView>(R.id.rv_user)
        rv.layoutManager = LinearLayoutManager(context)
        userAdapter = UserAdapter(userList)
        rv.adapter = userAdapter

        //数据库加载数据
        val dao = UserDao(requireContext())
        userList.clear()
        userList.addAll(dao.getAllUser())
        userAdapter.notifyDataSetChanged()

        //长按删除弹窗复用之前逻辑
        userAdapter.longClick = { selectUser->
            val dialog = android.app.Dialog(requireContext())
            dialog.setContentView(R.layout.dialog_delete)
            dialog.window?.setLayout(550, android.view.ViewGroup.LayoutParams.WRAP_CONTENT)
            dialog.findViewById<android.widget.Button>(R.id.btn_cancel).setOnClickListener { dialog.dismiss() }
            dialog.findViewById<android.widget.Button>(R.id.btn_confirm).setOnClickListener {
                dao.deleteUser(selectUser.account)
                userList.clear()
                userList.addAll(dao.getAllUser())
                userAdapter.notifyDataSetChanged()
                dialog.dismiss()
            }
            dialog.show()
        }
        return view
    }
}
```

### 3. MineFragment（我的：备忘录、设置、退出按钮）

```kotlin
import android.content.Intent
import android.os.Bundle
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.Button
import androidx.fragment.app.Fragment

class MineFragment : Fragment() {
    override fun onCreateView(inflater: LayoutInflater, container: ViewGroup?, savedInstanceState: Bundle?): View? {
        val view = inflater.inflate(R.layout.fragment_mine,container,false)
        //跳转备忘录
        view.findViewById<Button>(R.id.btn_memo).setOnClickListener {
            startActivity(Intent(context,MemoActivity::class.java))
        }
        //跳转设置
        view.findViewById<Button>(R.id.btn_setting).setOnClickListener {
            startActivity(Intent(context,SettingActivity::class.java))
        }
        //打开相册
        view.findViewById<Button>(R.id.btn_open_img).setOnClickListener {
            (activity as MainActivity).openAlbumFunc()
        }
        return view
    }
}
```

## 四、步骤3：MainActivity核心切换代码
把原来列表、相册权限代码保留，新增Fragment切换逻辑
```kotlin
import android.os.Bundle
import android.widget.RadioGroup
import androidx.appcompat.app.AppCompatActivity
import androidx.fragment.app.FragmentTransaction

class MainActivity : AppCompatActivity() {
    //三个碎片实例
    private val homeFrag = HomeFragment()
    private val userFrag = UserListFragment()
    private val mineFrag = MineFragment()

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        val rgTab = findViewById<RadioGroup>(R.id.rg_tab)

        //默认选中首页
        switchFragment(homeFrag)
        rgTab.check(R.id.rb_home)

        //底部切换监听
        rgTab.setOnCheckedChangeListener { _, checkedId ->
            when(checkedId){
                R.id.rb_home -> switchFragment(homeFrag)
                R.id.rb_user -> switchFragment(userFrag)
                R.id.rb_mine -> switchFragment(mineFrag)
            }
        }
    }

    //碎片替换通用方法
    private fun switchFragment(frag: androidx.fragment.app.Fragment){
        val ft:FragmentTransaction = supportFragmentManager.beginTransaction()
        ft.replace(R.id.fl_content,frag)
        ft.commit()
    }
```
