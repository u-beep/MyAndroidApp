## 二、步骤1：改造数据库User表（新增city字段）
> 数据库版本号+1，旧数据不影响，打开`MySQLiteHelper.kt`
```kotlin
class MySQLiteHelper(context: Context) : SQLiteOpenHelper(
    context,
    "user_db.db",
    null,
    2 // 版本由1改成2，触发onUpgrade升级
) {
    override fun onCreate(db: SQLiteDatabase) {
        //建表新增city TEXT
        val sql = """
            CREATE TABLE user_table (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                account TEXT UNIQUE,
                pwd TEXT,
                sex TEXT,
                hobby TEXT,
                city TEXT
            )
        """.trimIndent()
        db.execSQL(sql)
    }

    //版本升级，新增字段
    override fun onUpgrade(db: SQLiteDatabase, oldVersion: Int, newVersion: Int) {
        if(oldVersion == 1){
            db.execSQL("ALTER TABLE user_table ADD COLUMN city TEXT")
        }
    }
}
```

### 4.2 getAllUser读取城市
```kotlin
fun getAllUser(): MutableList<User> {
    val list = mutableListOf<User>()
    val cursor = db.query("user_table", null, null, null, null, null, null)
    while (cursor.moveToNext()) {
        val account = cursor.getString(1)
        val pwd = cursor.getString(2)
        val sex = cursor.getString(3)
        val hobby = cursor.getString(4)
        val city = cursor.getString(5)
        list.add(User(account, pwd, sex, hobby,city))
    }
    cursor.close()
    return list
}
```

## 六、步骤5：RegisterActivity绑定Spinner、获取选中城市
```kotlin
//1.定义城市数组数据源
val cityArr = arrayOf("北京","上海","广州","深圳","杭州","成都","重庆")
//绑定控件
val spCity = findViewById<Spinner>(R.id.sp_city)
var selectCity = cityArr[0] //默认第一个

//2.构建适配器
val adapter = android.widget.ArrayAdapter(
    this,
    android.R.layout.simple_spinner_dropdown_item,
    cityArr
)
spCity.adapter = adapter

//3.选中监听
spCity.onItemSelectedListener = object : android.widget.AdapterView.OnItemSelectedListener{
    override fun onItemSelected(p0: android.widget.AdapterView<*>?, p1: android.view.View?, pos: Int, p3: Long) {
        selectCity = cityArr[pos]
    }
    override fun onNothingSelected(p0: android.widget.AdapterView<*>?) {}
}
```
