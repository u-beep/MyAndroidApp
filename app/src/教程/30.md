## 三、封装OkHttp工具类 NetUtil.kt（全局统一请求）
```kotlin
import okhttp3.Call
import okhttp3.Callback
import okhttp3.OkHttpClient
import okhttp3.Request
import okhttp3.Response
import java.io.IOException

object NetUtil {
    private val client = OkHttpClient()

    //GET请求
    fun get(url:String,callback:NetCallBack){
        val req = Request.Builder().url(url).build()
        client.newCall(req).enqueue(object :Callback{
            override fun onFailure(call: Call, e: IOException) {
                callback.onError(e.message)
            }
            override fun onResponse(call: Call, response: Response) {
                val json = response.body?.string()
                callback.onSuccess(json)
            }
        })
    }

    //回调接口
    interface NetCallBack{
        fun onSuccess(json:String?)
        fun onError(msg:String?)
    }
}
```

## 五、Gson解析工具 GsonUtil.kt
```kotlin
import com.google.gson.Gson

object GsonUtil {
    private val gson = Gson()
    //JSON转对象
    fun <T> jsonToBean(json:String,clazz:Class<T>):T{
        return gson.fromJson(json,clazz)
    }
}
```

### NewsAdapter.kt
```kotlin
import android.view.LayoutInflater
import android.view.ViewGroup
import androidx.recyclerview.widget.RecyclerView
import com.example.myapplication.databinding.ItemNewsBinding

class NewsAdapter(var list:MutableList<NewsItem>):RecyclerView.Adapter<NewsAdapter.VH>(){
    inner class VH(val bind:ItemNewsBinding):RecyclerView.ViewHolder(bind.root)
    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): VH {
        val inflate = LayoutInflater.from(parent.context)
        return VH(ItemNewsBinding.inflate(inflate,parent,false))
    }
    override fun onBindViewHolder(holder: VH, position: Int) {
        val item = list[position]
        holder.bind.tvTitle.text = item.title
        holder.bind.tvInfo.text = "${item.source} | ${item.time}"
    }
    override fun getItemCount() = list.size
    fun refreshData(newList:List<NewsItem>){
        list.clear()
        list.addAll(newList)
        notifyDataSetChanged()
    }
}
```

### NewsFragment.kt 网络请求核心
```kotlin
import android.os.Bundle
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.Toast
import androidx.fragment.app.Fragment
import androidx.recyclerview.widget.LinearLayoutManager
import com.example.myapplication.databinding.FragmentNewsBinding

class NewsFragment : Fragment() {
    private lateinit var binding:FragmentNewsBinding
    private lateinit var adapter:NewsAdapter
    private var newsList = mutableListOf<NewsItem>()
    //免费测试新闻接口
    private val newsUrl = "https://xxxxxxx"

    override fun onCreateView(inflater: LayoutInflater, container: ViewGroup?, savedInstanceState: Bundle?): View {
        binding = FragmentNewsBinding.inflate(layoutInflater,container,false)
        adapter = NewsAdapter(newsList)
        binding.rvNews.adapter = adapter
        binding.rvNews.layoutManager = LinearLayoutManager(context)
        loadNews()
        return binding.root
    }

    private fun loadNews(){
        NetUtil.get(newsUrl,object :NetUtil.NetCallBack{
            override fun onSuccess(json: String?) {
                //子线程不能更新UI，切主线程
                activity?.runOnUiThread {
                    if(json.isNullOrEmpty()) return@runOnUiThread
                    val resp = GsonUtil.jsonToBean(json,NewsResp::class.java)
                    adapter.refreshData(resp.data)
                }
            }
            override fun onError(msg: String?) {
                activity?.runOnUiThread {
                    Toast.makeText(context,"请求失败:$msg",Toast.LENGTH_SHORT).show()
                }
            }
        })
    }
}
```

## 九、MVVM极简入门Demo（30天拓展）
### 分层：M(数据) V(XML+Activity) VM(中间桥梁)
1. Model：NewsDao/网络请求NetUtil（数据来源）
2. View：布局+Activity，只做页面展示，不写请求代码
3. ViewModel：接收View指令，调用Model拿数据，数据变更自动刷新页面

简易结构：
```
View(NewsFragment) → 点击刷新 → NewsViewModel → NetUtil(Model请求网络)
→ 数据返回ViewModel → 自动通知View刷新RecyclerView
```
> 优势：页面与数据逻辑彻底解耦，企业主流架构，30天之后进阶学习。
