# 🚀 第3步：编写设置页面逻辑
`SettingActivity.kt`

```kotlin
import android.content.Intent
import android.content.SharedPreferences
import android.os.Bundle
import android.widget.Button
import android.widget.Switch
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class SettingActivity : AppCompatActivity() {

    lateinit var sp: SharedPreferences
    lateinit var switchAuto: Switch

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_setting)

        switchAuto = findViewById(R.id.switch_auto)
        val btnLogout = findViewById<Button>(R.id.btn_logout)
        sp = getSharedPreferences("SETTING", MODE_PRIVATE)

        // 读取保存的开关状态，自动显示
        switchAuto.isChecked = sp.getBoolean("auto_login", false)

        // 开关状态改变时保存
        switchAuto.setOnCheckedChangeListener { button, isChecked ->
            sp.edit().putBoolean("auto_login", isChecked).apply()
            if (isChecked) {
                Toast.makeText(this, "已开启自动登录", Toast.LENGTH_SHORT).show()
            } else {
                Toast.makeText(this, "已关闭自动登录", Toast.LENGTH_SHORT).show()
            }
        }

        // 退出登录
        btnLogout.setOnClickListener {
            // 清除用户信息
            getSharedPreferences("USER_DATA", MODE_PRIVATE).edit().clear().apply()
            Toast.makeText(this, "退出成功", Toast.LENGTH_SHORT).show()
            startActivity(Intent(this, LoginActivity::class.java))
            finish()
        }
    }
}
```

# 🚀 第5步：实现 **打开APP自动登录**（核心）
打开 `LoginActivity.kt`，在 `onCreate` 最顶部加入：

```kotlin
// 读取设置
val spSetting = getSharedPreferences("SETTING", MODE_PRIVATE)
val autoLogin = spSetting.getBoolean("auto_login", false)

// 读取用户信息
val spUser = getSharedPreferences("USER_DATA", MODE_PRIVATE)
val account = spUser.getString("account", "")

// 如果开启了自动登录，并且有保存的账号，直接跳主页
if (autoLogin && !account.isNullOrEmpty()) {
    startActivity(Intent(this, MainActivity::class.java))
    finish()
}
```
